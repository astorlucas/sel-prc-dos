<!-- Esto basicamente es lo que sirve para correr desde consola y el include
			te cambia dependiendo de la nomenclatura que tu le des a tus clases test -->
				<artifactId>maven-surefire-plugin</artifactId>
				<version>2.22.2</version>
				<configuration>
					<includes>
						<include>**/*Test.java</include>
					</includes>
					<systemPropertyVariables>
						<application.properties>/application.properties</application.properties>
					</systemPropertyVariables>
				</configuration>
			</plugin>
